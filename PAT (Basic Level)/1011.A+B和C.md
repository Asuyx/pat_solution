## C++代码

```cpp
#include<cstdio>
int main(){
    long long a,b,c;
    int T;
    int i=1;
    scanf("%d",&T);
    while(T--){
        scanf("%lld%lld%lld",&a,&b,&c);
        if(a+b>c)
            printf("Case #%d: true\n",i);
        else
            printf("Case #%d: false\n",i);
        i++;
    }
    return 0;
}
```
