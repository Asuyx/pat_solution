//我的垃圾代码
#include<cstdio>
int main(){
    int a,da,b,db;
    int pa=0,pb=0;
    scanf("%d%d%d%d",&a,&da,&b,&db);
    int daa=da,dbb=db;
    while(a!=0){
        if(a%10==da){
            pa=pa+daa;
            daa*=10;
        }
        a/=10;
    }
    while(b!=0){
        if(b%10==db){
            pb=pb+dbb;
            dbb*=10;
        }
        b/=10;
    }
    printf("%d",pa+pb);
    return 0;
}
